___#ex0109 edit setting.py to include djoser
-----------------settings.py


| INSTALLED_APPS = [
| .. 'rest_framework',
| .. 'rest_framework.authtoken',
| .. 'djoser',
| ]


__# create new section named ' djoser'
| DJOSER = {
| .. "USER_ID_FIELD": "username",
| .. "LOGIN_FIELD":"email",
| }
